      # 1. 建立接口访问方法

      src/api/cats.js

      
import axios from "axios"
import Vue from 'vue';

// context 是store, 传递过来是为了可能需要访问state 或其他方法
// 引入vue是为了访问jsonp

// 这里把接口访问统一放在这里, 是为了统一接口的访问数据

// 比如每个接口返回的数据不一致,  我们在这里可以把数据都处理为一致的参数返回

// 如果写在store里 会导致store文件很长, 不便维护

export function api_getCats(api_url, param = {}) {
  return Vue.jsonp(api_url,param).then(
    (response) => {
      if(response.code == 200) {
        return Promise.resolve({code:0, data:response.data})
      } else {
        return Promise.reject({code:-1, data:response.msg})
      }
    }
  ).catch(
    (error) => {return Promise.reject({code:-2, data: error})}
  )
}

      # 2. vuex store的使用

      import Vue from 'vue';
import Vuex from 'vuex';
import {api_getCats}  from "@/api/cats" ;

Vue.use(Vuex);

export default new Vuex.Store({
  state: {
    cats: {},
    catsLoaded: false,
  },

  getters: {
    isCatsLoaded (state) {
      return state.catsLoaded;
    },

    // 返回所有的分类, 组件通过getters来获取分类,
    // 因为getters 可以进行映射, 访问更方便
    // 而且在getters 中可以进行数据的排序等, 但注意getters中不要对state进行改变
    // 这里直接对元数据返回
    allCats(state) {
      return state.cats
    }
  },

  mutations: {
    // 该方法只会在分类数据成功获取才会被调用
    setCats(state, cats_data) {
      state.cats = Object.assign({}, cats_data);
      state.catsLoaded = true;
    }
  },

  actions: {
    getCatsAction: function (context) {
      let cat_api_url = "http://pharmacy-backend.lindoit.com:8082/test-data/api-cat";
      let param = {};
      api_getCats(cat_api_url, param).then(response => {
        // console.log(response);
        if (response.code == 0) {
          // 分类数据获取成功, 才调用设置state的mutation方法
          context.commit("setCats", response.data);
        }
      })
    }
  },
});
